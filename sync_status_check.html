📊 Supabase 数据同步状态检查

实时查看本地和云端数据同步状态

💡 使用说明：
  1. 打开你的应用 index.html
  2. 按 F12 打开浏览器控制台
  3. 复制下面的 JavaScript 代码并粘贴到控制台执行
  4. 查看返回的同步状态报告

📝 检查代码（复制到浏览器控制台执行）：

// 检查 Supabase 数据同步状态
async function checkSyncStatus() {
  console.log('🔍 开始检查数据同步状态...\n');
  
  const results = {};
  
  // 检查日历数据
  const calendarLocal = Object.keys(store || {}).length;
  const { data: calendarCloud } = await window.mySupabase.from('calendar').select('*');
  const calendarCloudCount = calendarCloud?.length || 0;
  results.calendar = { local: calendarLocal, cloud: calendarCloudCount };
  
  // 检查纪念日
  const anniversaryLocal = (anniversaries || []).length;
  const { data: anniversaryCloud } = await window.mySupabase.from('anniversaries').select('*');
  const anniversaryCloudCount = anniversaryCloud?.length || 0;
  results.anniversaries = { local: anniversaryLocal, cloud: anniversaryCloudCount };
  
  // 检查倒计时
  const countdownLocal = (countdowns || []).length;
  const { data: countdownCloud } = await window.mySupabase.from('countdowns').select('*');
  const countdownCloudCount = countdownCloud?.length || 0;
  results.countdowns = { local: countdownLocal, cloud: countdownCloudCount };
  
  // 检查天气数据
  const weatherLocal = (weatherCities || []).length;
  const { data: weatherCloud } = await window.mySupabase.from('weather').select('*');
  const weatherCloudCount = weatherCloud?.length || 0;
  results.weather = { local: weatherLocal, cloud: weatherCloudCount };
  
  // 检查结果
  console.log('📊 数据同步状态报告：\n');
  console.table(results);
  
  let allSynced = true;
  for (const [table, counts] of Object.entries(results)) {
    const status = counts.local === counts.cloud ? '✅' : '❌';
    if (counts.local !== counts.cloud) allSynced = false;
    console.log(`${status} ${table}: 本地=${counts.local}, 云端=${counts.cloud}`);
  }
  
  console.log('\n' + (allSynced ? '✅ 所有数据已同步！' : '⚠️ 存在未同步的数据'));
  return results;
}

// 执行检查
checkSyncStatus();
🔄 数据同步机制说明：
  • 本地优先：所有操作先保存到 IndexedDB
  • 自动同步：启用 Supabase 后，自动同步到云端
  • 实时更新：通过 WebSocket 监听其他设备的更改
  • 防循环：使用 isRemoteUpdate 标志防止同步循环
  • 批量同步：纪念日、倒计时等使用批量同步提高性能
